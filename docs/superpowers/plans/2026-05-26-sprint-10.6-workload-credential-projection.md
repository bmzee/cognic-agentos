from pathlib import Path
from typing import Any

import pytest

from cognic_agentos.cli.validators.credentials import validate


# Baseline valid pack manifest with one [credentials.db_main] block.
# Tests override single fields to exercise each refusal mode.
_BASELINE: dict[str, Any] = {
    "pack": {"name": "test-pack", "version": "0.1.0"},
    "runtime": {"expected_workload_gid": 1000},
    "risk_tier": {"tier": "internal_write"},
    "credentials": {
        "db_main": {
            "vault_path": "database/creds/db-main",
            "expected_fields": ["username", "password"],
            "ttl_s": 900,
            "purpose_category": "application_database_read",
            "purpose_description": "Read-only database access.",
        },
    },
}


def _deep_merge(base: dict, overrides: dict) -> dict:
    """Recursive merge; overrides win at every nesting level."""
    import copy
    result = copy.deepcopy(base)
    for key, value in overrides.items():
        if isinstance(value, dict) and isinstance(result.get(key), dict):
            result[key] = _deep_merge(result[key], value)
        else:
            result[key] = value
    return result


def _assert_refuses_with(reason: str, overrides: dict, *, pack_path: Path = Path(".")) -> None:
    findings = validate(_deep_merge(_BASELINE, overrides), pack_path)
    refusal_reasons = {f.reason for f in findings if f.severity == "refusal"}
    assert reason in refusal_reasons, (
        f"Expected refusal reason {reason!r} in findings; got {refusal_reasons!r}"
    )


def _assert_passes(overrides: dict, *, pack_path: Path = Path(".")) -> None:
    findings = validate(_deep_merge(_BASELINE, overrides), pack_path)
    refusal_reasons = {f.reason for f in findings if f.severity == "refusal"}
    assert refusal_reasons == set(), (
        f"Expected zero refusals; got {refusal_reasons!r}"
    )

    def test_valid_lowercase_snake_case_passes(self):
        _assert_passes({"credentials": {"db_main_v2": _BASELINE["credentials"]["db_main"]}})

    def test_uppercase_refuses_with_logical_name_invalid_grammar(self):
        _assert_refuses_with(
            "credentials_logical_name_invalid_grammar",

class TestRiskTierCrossValidator:
    """Per spec §5.1 + [[feedback_dual_path_doctrine]]: one-validator-owns-each-refusal invariant."""

    @pytest.mark.parametrize("tier", [
        "customer_data_read", "customer_data_write", "payment_action",
        "regulator_communication", "cross_tenant", "high_risk_custom",
    ])
    def test_high_risk_tier_with_credentials_refuses_pre_13_5(self, tier):
        _assert_refuses_with(
            "credentials_risk_tier_not_permitted_pre_13_5",
            {"risk_tier": {"tier": tier}},
        )

    def test_missing_risk_tier_suppresses_credentials_risk_check(self):
        # Risk-tier validator emits its existing missing-risk-tier finding;
        # credentials validator suppresses its own risk-tier check (which
        # would otherwise produce a misleading false positive).
        manifest = _deep_merge(_BASELINE, {"risk_tier": None})
        del manifest["risk_tier"]
        findings = validate(manifest, Path("."))
        credentials_risk_reasons = {
            f.reason for f in findings
            if f.reason == "credentials_risk_tier_not_permitted_pre_13_5"
        }
        assert credentials_risk_reasons == set(), (
            "credentials validator must suppress its risk-tier check when "
            "the upstream risk-tier validator's missing-risk-tier finding fires"
        )

- Opt-out (env unset): module-level pytest.skip BEFORE optional imports → silent skip
  regardless of extras availability
- Opt-in (env set): plain import → missing extras → ImportError → pytest collection error

Subprocess isolation required because the parent test process has already
imported hvac/aiodocker/kubernetes_asyncio (they ARE installed). sys.path
shim shadows them with fake-raising modules.
"""
import os
import subprocess
import sys
import textwrap
from pathlib import Path


def _build_shim_dir(tmp_path: Path, module_names: list[str]) -> Path:
    shim = tmp_path / "shim"
    shim.mkdir()
    for name in module_names:
        pkg = shim / name
        pkg.mkdir()
        (pkg / "__init__.py").write_text(
            f"raise ImportError('Sprint 10.6 regression shim — "
            f"{name} intentionally unimportable to verify Z3/Z4 fail-loud contract')",
            encoding="utf-8",
        )
    return shim


class TestZ3ImportContract:
    def test_z3_opted_in_fails_loud_when_hvac_missing(self, tmp_path):
        shim = _build_shim_dir(tmp_path, ["hvac", "aiodocker"])
        script = textwrap.dedent(f"""
            import sys
            sys.path.insert(0, {str(shim)!r})
            try:
                import tests.integration.sandbox.test_z3_docker_credential_projection  # noqa
            except ImportError as e:
                print(f"FAIL_LOUD_OK: {{e}}", flush=True)
                sys.exit(7)
            print("SILENT_SKIP_BUG", flush=True)
            sys.exit(0)
        """)
        env = {**os.environ, "COGNIC_RUN_DOCKER_CREDENTIAL_PROJECTION_INTEGRATION": "1"}
        result = subprocess.run(
            [sys.executable, "-c", script],
            env=env, capture_output=True, text=True,
            cwd=str(Path(__file__).resolve().parents[2]),
            timeout=60,
        )
        assert result.returncode == 7, f"Expected fail-loud (exit 7); got {result.returncode}\nstdout={result.stdout!r}\nstderr={result.stderr!r}"
        assert "FAIL_LOUD_OK" in result.stdout

    def test_z3_opted_out_silent_skip_even_when_extras_missing(self, tmp_path):
        """Opt-out should skip BEFORE optional imports — extras availability irrelevant."""
        shim = _build_shim_dir(tmp_path, ["hvac", "aiodocker"])
        script = textwrap.dedent(f"""
            import pytest
            import sys
            sys.path.insert(0, {str(shim)!r})
            try:
                import tests.integration.sandbox.test_z3_docker_credential_projection  # noqa
            except pytest.skip.Exception:
                print("CASUAL_SKIP_OK", flush=True)
                sys.exit(7)
            except ImportError as e:
                print(f"UNEXPECTED_IMPORT_ERROR: {{e}}", flush=True)
                sys.exit(8)
            print("UNEXPECTED_OK", flush=True)
            sys.exit(0)
        """)
        env = {**os.environ}
        env.pop("COGNIC_RUN_DOCKER_CREDENTIAL_PROJECTION_INTEGRATION", None)
        result = subprocess.run(
            [sys.executable, "-c", script],
            env=env, capture_output=True, text=True,
            cwd=str(Path(__file__).resolve().parents[2]),
            timeout=60,
        )
        assert result.returncode == 7
        assert "CASUAL_SKIP_OK" in result.stdout
