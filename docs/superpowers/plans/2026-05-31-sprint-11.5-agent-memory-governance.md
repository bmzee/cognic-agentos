# Sprint 11.5 — Agent Memory Governance Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `core/memory/` — the governed memory substrate (7-op MemoryAPI + 3 block ops, 3 tiers, default-deny `long_term`, per-write/recall enforcement, DLP seed, chain-linked audit) per ADR-019, DI-tested and not yet harness-injected.

**Architecture:** A construction-bound `MemoryAPI` (identity + served-subject fixed at build time, never per-call) routes every operation through an ordered enforcement gate (kill-switch → sub-agent guard → cross-subject → tier-admit → DLP → purpose → consent → retention → approval-transitional → audit) before touching a `MemoryAdapter`. Memory blocks are singleton `long_term` projections, not a fourth tier. Default-deny lives in two OPA/Rego bundles consumed via the Sprint-4 `OPAEngine`. Memory events ride the existing `decision_history` chain (no `core/canonical.py` / `decision_history.py` schema change; value never enters the chain — digest only).

**Tech Stack:** Python 3.12 (`uv run`), SQLAlchemy 2 async + Alembic, OPA/Rego (`core/policy/engine.OPAEngine`), Pydantic-settings, qdrant + embedding adapters (vector recall), Redis (scratch tier), pytest + pytest-asyncio. No new runtime dependencies (DLP seed is stdlib `re` + pure-Python checksums + a gazetteer data file).

**Scope:** 11.5a (T0–T12 + Z1a) in full TDD detail below. 11.5b (T13–T20 + Z1b) is a structured outline with a VALVE-CHECK deferral footer (Sprint 10.5 / Sprint 11 precedent).

"""Sprint 11.5a — memory tiers, block kinds, subject vocabulary, refusal taxonomy.

core/ stop-rule per AGENTS.md (Memory governance enforcement, ADR-019).
The DataClass / Purpose / RESTRICTED_DATA_CLASSES copies below are a
DELIBERATE inline mirror of cli/_governance_vocab — core/ MUST NOT import
cli/* at runtime (architectural arrow runs cli -> core). Lockstep is pinned
test-only by tests/unit/core/memory/test_vocab_drift.py
(feedback_drift_detector_test_only_no_runtime_import).
"""
from __future__ import annotations
import dataclasses
from typing import Literal

MemoryTier = Literal["scratch", "task", "long_term"]
BlockKind = Literal["persona", "user_profile", "agent_notes"]

# --- Local copy of cli/_governance_vocab (drift-pinned test-only) ---
DataClass = Literal[
    "public", "internal", "customer_pii", "payment_data", "credentials",
    "regulator_communication", "audit_trail", "model_inputs", "model_outputs",
]
Purpose = Literal[
    "transaction_processing", "regulatory_reporting", "fraud_detection",
    "customer_support", "audit_evidence", "operational_telemetry",
]
RESTRICTED_DATA_CLASSES: frozenset[str] = frozenset(
    {"customer_pii", "payment_data", "credentials", "regulator_communication"}
)

MemoryRefusalReason = Literal[
    "memory_write_frozen",
    "memory_subagent_durable_access_refused",
    "memory_long_term_write_denied",
    "memory_dlp_undeclared_restricted_class",
    "memory_purpose_not_declared",
    "memory_consent_required",
    "memory_consent_invalid",
    "memory_approval_engine_not_available",
    "memory_recall_capability_missing",
    "memory_cross_subject_access_refused",
    "memory_purpose_mismatch",
]

@dataclasses.dataclass(frozen=True, slots=True)
class SubjectRef:
    kind: Literal["human", "agent"]
    id: str
    def __post_init__(self) -> None:
        if not self.id:
            raise ValueError("SubjectRef.id must be non-empty (no tenant-wide/unscoped memory)")
    @property
    def canonical(self) -> str:
        return f"{self.kind}:{self.id}"

class MemoryOperationRefused(Exception):
    """Typed refusal carrying ONLY the wire-public closed-enum reason."""
    def __init__(self, reason: MemoryRefusalReason) -> None:
        super().__init__(reason)
        self.reason: MemoryRefusalReason = reason

def _luhn_ok(digits: str) -> bool:
    s, alt = 0, False
    for ch in reversed(digits):
        d = ord(ch) - 48
        if alt:
            d *= 2
            if d > 9:
                d -= 9
        s += d
        alt = not alt
    return s % 10 == 0

def _iban_ok(iban: str) -> bool:
    s = iban[4:] + iban[:4]
    n = "".join(str(ord(c) - 55) if c.isalpha() else c for c in s)
    return int(n) % 97 == 1
```

from __future__ import annotations
import dataclasses, datetime as dt, hashlib
from cognic_agentos.core.memory.tiers import SubjectRef, MemoryOperationRefused
from cognic_agentos.core.decision_history import DecisionHistoryStore, DecisionRecord

@dataclasses.dataclass(frozen=True, slots=True)
class ConsentToken:
    subject_ref: str                 # canonical SubjectRef string
    data_classes: frozenset[str]
    issued_at: dt.datetime
    expires_at: dt.datetime
    signature: str

class ConsentValidator:
    def __init__(self, *, audit: DecisionHistoryStore) -> None:
        self._audit = audit

    async def validate(self, token: ConsentToken | None, *, served_subject: SubjectRef,
                       restricted_declared: frozenset[str], tenant_id: str, actor_id: str) -> None:
        if not restricted_declared:
            return                                            # no restricted class → no consent needed
        if token is None:
            raise MemoryOperationRefused("memory_consent_required")
        now = dt.datetime.now(dt.timezone.utc)                # wall-clock expiry; not a chain timestamp
        if (token.expires_at <= now
                or token.subject_ref != served_subject.canonical
                or not restricted_declared <= token.data_classes):
            raise MemoryOperationRefused("memory_consent_invalid")
        digest = hashlib.sha256(f"{token.subject_ref}|{token.signature}".encode()).hexdigest()
        await self._audit.append(DecisionRecord(
            decision_type="memory.consent",
            request_id=f"memory-consent-{digest[:16]}",
            payload={"subject_ref": token.subject_ref, "data_classes": sorted(token.data_classes),
                     "consent_token_digest": digest},        # digest only — never the raw token
            actor_id=actor_id, tenant_id=tenant_id, iso_controls=("A.7.4", "A.8.2"),
        ))
```
